#include <iostream>
using namespace std;
int main()
{
    int tab[10][10],i,j;

    i = 0;
    j = 1;

   


    for (i = 0; i < 10;i++) {

        for (j = 1; j < 10;j++) {


            if(i == j){

            tab[i][j] = 1;
            }
            else {
                tab[i][j] = 0;
            }

        }


    }
    for (i = 0; i < 10; i++) {

        for (j = 1; j < 10; j++) {

            cout << tab[i][j] << " ";


        }
        cout << endl;

    }
    cout << endl;

}
